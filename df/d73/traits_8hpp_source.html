264// Old GCC versions have a bug with stack aligned to a 16-byte boundary as
265// mentioned at https://en.wikipedia.org/wiki/X86_calling_conventions. We can't
266// know for sure if this bug is present or not, as it depends on the
267// compiler -mpreffered-stack-boundary parameter, as I understand it.
268
272constexpr auto kStackAlignment = sizeof(mywr::address_t);
273
278template <typename T>
279constexpr auto align_parameter_size() {
280 if constexpr (std::is_reference_v<T> || std::is_pointer_v<T>)
281 return kStackAlignment;
282
283 constexpr auto remainder = sizeof(T) % kStackAlignment;
284 return remainder == 0 ? sizeof(T) : sizeof(T) + kStackAlignment - remainder;
285}
286
291template <typename T, typename... Ts>
292constexpr auto align_parameters_size() {
293 if constexpr (sizeof...(Ts) > 0)
294 return align_parameter_size<T>() + align_parameters_size<Ts...>();
295 else
296 return align_parameter_size<T>();
297}
298
302template <typename T, typename... Ts>
303constexpr auto sizeof_parameters_size() {
304 if constexpr (sizeof...(Ts) > 0)
305 return sizeof(T) + sizeof_parameters_size<Ts...>();
306 else
307 return sizeof(T);
308}
