118static memory_prot::Enum to_protection_constant(const std::uint32_t protect) {
119#if defined(MYWR_WINDOWS)
120 switch (protect) {
121 case PAGE_NOACCESS:
122 return memory_prot::kNoAccess;
123 case PAGE_READONLY:
124 return memory_prot::kReadOnly;
125 case PAGE_READWRITE:
126 return memory_prot::kReadWrite;
127 case PAGE_WRITECOPY:
128 return memory_prot::kWriteCopy;
129 case PAGE_EXECUTE:
130 return memory_prot::kExecute;
131 case PAGE_EXECUTE_READ:
132 return memory_prot::kExecuteRead;
133 case PAGE_EXECUTE_READWRITE:
134 return memory_prot::kExecuteReadWrite;
135 case PAGE_EXECUTE_WRITECOPY:
136 return memory_prot::kExecuteWriteCopy;
137 default:
138 return memory_prot::kUnknown;
139 }
140#elif defined(MYWR_UNIX) && !defined(MYWR_FEATURE_NO_MPROTECT)
141 switch (protect) {
142 case PROT_NONE:
143 return memory_prot::kNoAccess;
144 case PROT_READ:
145 return memory_prot::kRead;
146 case PROT_WRITE:
147 return memory_prot::kWrite;
148 case PROT_EXEC:
149 return memory_prot::kExecute;
150 case PROT_READ | PROT_WRITE:
151 return memory_prot::kReadWrite;
152 case PROT_READ | PROT_EXEC:
153 return memory_prot::kExecuteRead;
154 case PROT_READ | PROT_WRITE | PROT_EXEC:
155 return memory_prot::kExecuteReadWrite;
156 default:
157 return memory_prot::kUnknown;
158 }
159#else
160 return memory_prot::kRead | memory_prot::kWrite | memory_prot::kExecute;
161#endif
162}

245static memory_prot::Enum get_protect(const address& target) {
246#if defined(MYWR_WINDOWS)
247 MEMORY_BASIC_INFORMATION mbi{};
248 if (!VirtualQuery(target, &mbi, sizeof(mbi)))
249 return memory_prot::kUnknown;
250
251 return to_protection_constant(mbi.Protect);
252#elif defined(MYWR_UNIX) && !defined(MYWR_FEATURE_NO_MPROTECT)
253 // Source:
254 // https://github.com/kin4stat/kthook/blob/main/include/kthook/x86_64/kthook_x86_64_detail.hpp#L591
255 struct map_info {
256 std::uintptr_t start;
257 std::uintptr_t end;
258
259 unsigned prot;
260 };
261
262 std::vector<map_info> result;
263
264 #ifdef __FreeBSD__
265 std::ifstream proc_maps{"/proc/curproc/map"};
266 #else
267 std::ifstream proc_maps{"/proc/self/maps"};
268 #endif
269 std::string line;
270
271 while (std::getline(proc_maps, line)) {
272 map_info parse_result{};
273
274 auto start = 0u;
275 auto i = 0u;
276 while (line[i] != '-') {
277 ++i;
278 }
279
280 std::from_chars(&line[start], &line[i], parse_result.start, 16);
281
282 start = ++i;
283 while (line[i] != '\t' && line[i] != ' ') {
284 ++i;
285 }
286 std::from_chars(&line[start], &line[i], parse_result.end, 16);
287
288 #ifdef __FreeBSD__
289 while (line[i] != '\t' && line[i] != ' ') {
290 ++i;
291 }
292 ++i; // skip
293 while (line[i] != '\t' && line[i] != ' ') {
294 ++i;
295 }
296 ++i; // skip
297 while (line[i] != '\t' && line[i] != ' ') {
298 ++i;
299 }
300 ++i; // skip
301 #endif
302
303 start = ++i;
304 while (line[i] != '\t' && line[i] != ' ') {
305 ++i;
306 }
307
308 if (line[start++] == 'r')
309 parse_result.prot |= PROT_READ;
310 if (line[start++] == 'w')
311 parse_result.prot |= PROT_WRITE;
312 if (line[start++] == 'x')
313 parse_result.prot |= PROT_EXEC;
314 result.push_back(parse_result);
315 }
316
317 address_t address = target.value();
318 for (auto& mi : result) {
319 if (address >= mi.start && address <= mi.end) {
320 return to_protection_constant(mi.prot);
321 }
322 }
323
324 return memory_prot::kUnknown;
325#else
326 return memory_prot::kUnknown;
327#endif
328}

343static memory_prot::Enum set_protect(const address& target,
344 const std::size_t size,
345 const memory_prot::Enum protect) {
346#if defined(MYWR_WINDOWS)
347 DWORD old_protect{};
348 if (!VirtualProtect(target,
349 size,
350 from_protection_constant(protect),
351 &old_protect))
352 return memory_prot::kUnknown;
353
354 return to_protection_constant(old_protect);
355#elif defined(MYWR_UNIX) && !defined(MYWR_FEATURE_NO_MPROTECT)
356 address_t address = target.value();
357 address_t aligned_address = address & ~(sysconf(_SC_PAGE_SIZE) - 1u);
358 size_t aligned_size = size + (address - aligned_address);
359
360 // Retrieve old protect on UNIX systems.
361 memory_prot::Enum old_protect = get_protect(address);
362
363 // Set new protect.
364 if (mprotect(reinterpret_cast<void*>(aligned_address),
365 aligned_size,
366 from_protection_constant(protect)) != 0)
367 return memory_prot::kUnknown;
368
369 // And return old protect.
370 return old_protect;
371#else
372 return memory_prot::kUnknown;
373#endif
374}
