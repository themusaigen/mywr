259 template <typename T,
260 typename = std::enable_if_t<
261 std::is_integral_v<T> || std::is_floating_point_v<T> ||
262 std::is_same_v<T, double> || std::is_same_v<T, long double>,
263 T>>
264 void grab_number(T& value, int radix = 10) {
265 std::from_chars(&m_line[m_scope], &m_line[m_cursor], value, radix);
266 }
267
271 std::string grab_string() {
272 return m_line.substr(m_scope, m_cursor - m_scope);
273 }
274
278 char now() const { return m_line[m_cursor]; }
279
283 char next() { return m_line[m_cursor++]; }
284
288 void next_line() {
289 if (!fail())
290 std::getline(m_file, m_line);
291
292 m_scope = 0;
293 m_cursor = 0;
294 }

340static void parse_maps(std::vector<memory_region>& regions) {
341#if defined(MYWR_UNIX)
342 #if defined(__FreeBSD__)
343 constexpr auto kProcMapsPath = "/proc/curproc/map";
344 #else
345 constexpr auto kProcMapsPath = "/proc/self/maps";
346 #endif
347 constexpr auto kVdso = "[vdso]";
348 constexpr auto kVvar = "[vvar]";
349 constexpr auto kStack = "[stack]";
350 constexpr auto kThreadstack = "[stack:";
351 constexpr auto kAnon = "[anon:";
352 constexpr auto kAnonShmem = "[anon_shmem:";
353 constexpr auto kHeap = "[heap]";
354
355 parser parser{kProcMapsPath};
356
357 while (!parser.eof()) {
358 memory_region region{};
359
363 parser.scope();
364 parser.next_until('-');
365 parser.grab_number(region.begin, 16);
366 parser.next();
367
368 parser.scope();
369 parser.next_until_space();
370 parser.grab_number(region.end, 16);
371 parser.next();
372
376 if (parser.next() == 'r')
377 region.permissions |= PROT_READ;
378
379 if (parser.next() == 'w')
380 region.permissions |= PROT_WRITE;
381
382 if (parser.next() == 'x')
383 region.permissions |= PROT_EXEC;
384
388 char ps = parser.next();
389 if (ps == 'p')
390 region.is_private = true;
391 else if (ps == 's')
392 region.is_shared = true;
393
397 parser.next_until_space();
398 parser.next();
399
403 parser.scope();
404 parser.next_until_space();
405 parser.grab_number(region.offset, 16);
406
410 parser.next_until_space();
411 parser.next();
412
416 parser.scope();
417 parser.next_until(':');
418 parser.grab_number(region.dev_major);
419 parser.next();
420
421 parser.scope();
422 parser.next_until_space();
423 parser.grab_number(region.dev_minor);
424 parser.next();
425
429 parser.scope();
430 parser.next_until_space();
431 parser.grab_number(region.inode);
432
436 parser.next_until_any_char();
437 parser.scope();
438 parser.next_to_eol();
439
440 region.pathname = std::move(parser.grab_string());
441
442 #if defined(MYWR_FEATURE_PROCFS_PATHTYPE_DEDUCTION)
443 if (region.pathname == kStack)
444 region.pathtype = memory_region::kStack;
445 else if (region.pathname == kHeap)
446 region.pathtype = memory_region::kHeap;
447 else if (region.pathname == kVvar)
448 region.pathtype = memory_region::kVvar;
449 else if (region.pathname == kVdso)
450 region.pathtype = memory_region::kVdso;
451 else if (!region.pathname.empty())
452 region.pathtype = memory_region::kFile;
453 else if (region.pathname.find(kThreadstack) != std::string::npos)
454 region.pathtype = memory_region::kThreadstack;
455 else if (region.pathname.find(kAnon) != std::string::npos)
456 region.pathtype = memory_region::kAnon;
457 else if (region.pathname.find(kAnonShmem) != std::string::npos)
458 region.pathtype = memory_region::kAnonShmem;
459 else
460 region.pathtype = memory_region::kNone;
461 #endif
462
463 regions.push_back(std::move(region));
464
465 parser.next_line();
466 }
467#endif
468}
