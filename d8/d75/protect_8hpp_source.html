118static memory_prot::Enum to_protection_constant(const std::uint32_t protect) {
119#if defined(MYWR_WINDOWS)
120 switch (protect) {
121 case PAGE_NOACCESS:
122 return memory_prot::kNoAccess;
123 case PAGE_READONLY:
124 return memory_prot::kReadOnly;
125 case PAGE_READWRITE:
126 return memory_prot::kReadWrite;
127 case PAGE_WRITECOPY:
128 return memory_prot::kWriteCopy;
129 case PAGE_EXECUTE:
130 return memory_prot::kExecute;
131 case PAGE_EXECUTE_READ:
132 return memory_prot::kExecuteRead;
133 case PAGE_EXECUTE_READWRITE:
134 return memory_prot::kExecuteReadWrite;
135 case PAGE_EXECUTE_WRITECOPY:
136 return memory_prot::kExecuteWriteCopy;
137 default:
138 return memory_prot::kUnknown;
139 }
140#elif defined(MYWR_UNIX) && !defined(MYWR_FEATURE_NO_MPROTECT)
141 switch (protect) {
142 case PROT_NONE:
143 return memory_prot::kNoAccess;
144 case PROT_READ:
145 return memory_prot::kRead;
146 case PROT_WRITE:
147 return memory_prot::kWrite;
148 case PROT_EXEC:
149 return memory_prot::kExecute;
150 case PROT_READ | PROT_WRITE:
151 return memory_prot::kReadWrite;
152 case PROT_READ | PROT_EXEC:
153 return memory_prot::kExecuteRead;
154 case PROT_READ | PROT_WRITE | PROT_EXEC:
155 return memory_prot::kExecuteReadWrite;
156 default:
157 return memory_prot::kUnknown;
158 }
159#else
160 return memory_prot::kRead | memory_prot::kWrite | memory_prot::kExecute;
161#endif
162}

181static uint32_t from_protection_constant(const memory_prot::Enum protect) {
182#if defined(MYWR_WINDOWS)
183 switch (protect) {
184 case memory_prot::kNoAccess:

186 case memory_prot::kReadOnly:
187 return PAGE_READONLY;
188 case memory_prot::kReadWrite:
189 return PAGE_READWRITE;
190 case memory_prot::kWriteCopy:

192 case memory_prot::kExecute:
193 return PAGE_EXECUTE;
194 case memory_prot::kExecuteRead:
195 return PAGE_EXECUTE_READ;
196 case memory_prot::kExecuteReadWrite:
197 return PAGE_EXECUTE_READWRITE;
198 case memory_prot::kExecuteWriteCopy:

203#elif defined(MYWR_UNIX) && !defined(MYWR_FEATURE_NO_MPROTECT)

207 case memory_prot::kReadOnly:
208 return PROT_READ;
209 case memory_prot::kRead:
210 return PROT_READ;
211 case memory_prot::kWrite:
212 return PROT_WRITE;
213 case memory_prot::kExecute:

217 case memory_prot::kExecuteReadWrite:

245static memory_prot::Enum get_protect(const address& target) {
246#if defined(MYWR_WINDOWS)
247 MEMORY_BASIC_INFORMATION mbi{};
248 if (!VirtualQuery(target, &mbi, sizeof(mbi)))
249 return memory_prot::kUnknown;
250
251 return to_protection_constant(mbi.Protect);
252#elif defined(MYWR_UNIX) && !defined(MYWR_FEATURE_NO_MPROTECT)
256 std::vector<procfs::memory_region> regions;
257 procfs::parse_maps(regions);
258
262 address_t address = target.value();
263
267 for (auto& region : regions)
268 if (address >= region.begin && address <= region.end)
269 return to_protection_constant(region.permissions);
270
271 return memory_prot::kUnknown;
272#else
273 return memory_prot::kUnknown;
274#endif
275}

290static memory_prot::Enum set_protect(const address& target,
291 const std::size_t size,
292 const memory_prot::Enum protect) {
293#if defined(MYWR_WINDOWS)
294 DWORD old_protect{};
295 if (!VirtualProtect(target,
296 size,
297 from_protection_constant(protect),
298 &old_protect))
299 return memory_prot::kUnknown;
300
301 return to_protection_constant(old_protect);
302#elif defined(MYWR_UNIX) && !defined(MYWR_FEATURE_NO_MPROTECT)
303 address_t address = target.value();
304 address_t aligned_address = address & ~(sysconf(_SC_PAGE_SIZE) - 1u);
305 size_t aligned_size = size + (address - aligned_address);
306
307 // Retrieve old protect on UNIX systems.
308 memory_prot::Enum old_protect = get_protect(address);
309
310 // Set new protect.
311 if (mprotect(reinterpret_cast<void*>(aligned_address),
312 aligned_size,
313 from_protection_constant(protect)) != 0)
314 return memory_prot::kUnknown;
315
316 // And return old protect.
317 return old_protect;
318#else
319 return memory_prot::kUnknown;
320#endif
321}
