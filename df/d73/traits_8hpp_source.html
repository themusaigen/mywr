276// Old GCC versions have a bug with stack aligned to a 16-byte boundary as
277// mentioned at https://en.wikipedia.org/wiki/X86_calling_conventions. We can't
278// know for sure if this bug is present or not, as it depends on the
279// compiler -mpreffered-stack-boundary parameter, as I understand it.
280
284constexpr auto kStackAlignment = sizeof(mywr::address_t);
285
290template <typename T>
291constexpr auto align_parameter_size() {
292 if constexpr (std::is_reference_v<T> || std::is_pointer_v<T>)
293 return kStackAlignment;
294
295 constexpr auto remainder = sizeof(T) % kStackAlignment;
296 return remainder == 0 ? sizeof(T) : sizeof(T) + kStackAlignment - remainder;
297}
298
303template <typename T, typename... Ts>
304constexpr auto align_parameters_size() {
305 if constexpr (sizeof...(Ts) > 0)
306 return align_parameter_size<T>() + align_parameters_size<Ts...>();
307 else
308 return align_parameter_size<T>();
309}
310
314template <typename T, typename... Ts>
315constexpr auto sizeof_parameters_size() {
316 if constexpr (sizeof...(Ts) > 0)
317 return sizeof(T) + sizeof_parameters_size<Ts...>();
318 else
319 return sizeof(T);
320}
